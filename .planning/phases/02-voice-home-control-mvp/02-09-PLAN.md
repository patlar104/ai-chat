```kotlin
@HiltViewModel
class TasksViewModel @Inject constructor(
    private val reminderRepository: ReminderRepository,
    private val reminderScheduler: ReminderScheduler,
) : ViewModel() {

    val pendingReminders: StateFlow<List<Reminder>> = reminderRepository.observePending()
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), emptyList())

    fun cancelReminder(id: Long) {
        viewModelScope.launch {
            reminderScheduler.cancel(id)
            reminderRepository.updateStatus(id, ReminderStatus.CANCELLED)
        }
    }
}
```

```kotlin
@Composable
fun TasksScreen(viewModel: TasksViewModel = hiltViewModel()) {
    val reminders by viewModel.pendingReminders.collectAsStateWithLifecycle()
    var reminderToDelete by remember { mutableStateOf<Reminder?>(null) }

    Column(modifier = Modifier.fillMaxSize().background(MaterialTheme.colorScheme.background)) {
        Text(
            "Upcoming Reminders",
            style = MaterialTheme.typography.headlineSmall,
            color = MaterialTheme.colorScheme.onSurface,
            modifier = Modifier.padding(horizontal = Spacing.md, vertical = Spacing.md),
        )

        if (reminders.isEmpty()) {
            Box(Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
                Column(horizontalAlignment = Alignment.CenterHorizontally) {
                    Text("No reminders scheduled", style = headlineSmall, color = onSurface)
                    Spacer(Modifier.height(Spacing.sm))
                    Text("Say \"remind me to...\" to create one.", style = bodyLarge, color = onSurfaceVariant)
                }
            }
        } else {
            LazyColumn(
                contentPadding = PaddingValues(horizontal = Spacing.md),
                verticalArrangement = Arrangement.spacedBy(Spacing.sm),
            ) {
                items(reminders, key = { it.id }) { reminder ->
                    ReminderRow(
                        reminder = reminder,
                        onDeleteClick = { reminderToDelete = reminder },
                    )
                }
            }
        }
    }

    // Delete confirmation dialog
    reminderToDelete?.let { reminder ->
        AlertDialog(
            onDismissRequest = { reminderToDelete = null },
            title = { Text("Cancel Reminder") },
            text = { Text("Cancel this reminder? This cannot be undone.") },
            confirmButton = {
                TextButton(
                    onClick = {
                        viewModel.cancelReminder(reminder.id)
                        reminderToDelete = null
                    },
                    colors = ButtonDefaults.textButtonColors(contentColor = MaterialTheme.colorScheme.error),
                ) { Text("Cancel Reminder") }
            },
            dismissButton = {
                TextButton(onClick = { reminderToDelete = null }) { Text("Keep") }
            },
        )
    }
}

@Composable
private fun ReminderRow(reminder: Reminder, onDeleteClick: () -> Unit) {
    // surfaceContainer card, md padding
    // Description in bodyLarge (16sp)
    // Scheduled time in labelMedium (14sp), onSurfaceVariant
    // Delete icon button (error color, 48dp touch target) aligned to end
    Card(
        colors = CardDefaults.cardColors(containerColor = MaterialTheme.colorScheme.surfaceContainer),
        modifier = Modifier.fillMaxWidth(),
    ) {
        Row(
            modifier = Modifier.padding(Spacing.md).fillMaxWidth(),
            verticalAlignment = Alignment.CenterVertically,
        ) {
            Column(modifier = Modifier.weight(1f)) {
                Text(reminder.description, style = MaterialTheme.typography.bodyLarge, color = MaterialTheme.colorScheme.onSurface)
                Spacer(Modifier.height(Spacing.xs))
                Text(
                    formatTime(reminder.triggerTimeMs),
                    style = MaterialTheme.typography.labelMedium,
                    color = MaterialTheme.colorScheme.onSurfaceVariant,
                )
            }
            IconButton(onClick = onDeleteClick) {
                Icon(Icons.Filled.Delete, contentDescription = "Delete reminder", tint = MaterialTheme.colorScheme.error)
            }
        }
    }
}

    java.text.SimpleDateFormat("EEE, h:mm a", java.util.Locale.getDefault()).format(java.util.Date(timeMs))
```

**SettingsViewModel.kt:**
```kotlin
@HiltViewModel
class SettingsViewModel @Inject constructor(
    private val settingsRepository: SettingsRepository,
    private val ttsManager: TextToSpeechManager,
    private val haAuthInterceptor: HaAuthInterceptor,
    private val baseUrlInterceptor: BaseUrlInterceptor,
) : ViewModel() {

    val haServerUrl: StateFlow<String> = settingsRepository.haServerUrl
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), "")

    val haAccessTokenPresent: StateFlow<Boolean> = settingsRepository.haAccessToken
        .map { it != null }
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), false)

    val ttsVoiceName: StateFlow<String?> = settingsRepository.ttsVoiceName
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), null)

    val privacyModeEnabled: StateFlow<Boolean> = settingsRepository.privacyModeEnabled
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), false)

    val backgroundAutomationEnabled: StateFlow<Boolean> = settingsRepository.backgroundAutomationEnabled
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), true)

    val availableVoices: List<String>
        get() = ttsManager.getAvailableVoices().map { it.name }

    // Keep interceptors updated with current settings
    init {
        viewModelScope.launch {
            settingsRepository.haServerUrl.collect { url ->
                if (url.isNotBlank()) {
                    try {
                        baseUrlInterceptor.baseUrl = okhttp3.HttpUrl.Builder()
                            .scheme(if (url.startsWith("https")) "https" else "http")
                            .host(url.removePrefix("http://").removePrefix("https://").trimEnd('/').split(":").first())
                            .port(url.removePrefix("http://").removePrefix("https://").trimEnd('/').split(":").getOrNull(1)?.toIntOrNull() ?: if (url.startsWith("https")) 443 else 80)
                            .build()
                    } catch (_: Exception) { /* Invalid URL — ignore until user fixes */ }
                }
            }
        }
        viewModelScope.launch {
            settingsRepository.haAccessToken.collect { token ->
                haAuthInterceptor.token = token
            }
        }
    }

    fun saveHaServerUrl(url: String) {
        viewModelScope.launch { settingsRepository.setHaServerUrl(url) }
    }

    fun saveHaAccessToken(token: String) {
        viewModelScope.launch { settingsRepository.setHaAccessToken(token) }
    }

    fun clearHaAccessToken() {
        viewModelScope.launch { settingsRepository.clearHaAccessToken() }
    }

    fun saveTtsVoice(voiceName: String) {
        viewModelScope.launch {
            settingsRepository.setTtsVoiceName(voiceName)
            ttsManager.setVoice(voiceName)
        }
    }

    fun setPrivacyMode(enabled: Boolean) {
        viewModelScope.launch { settingsRepository.setPrivacyModeEnabled(enabled) }
    }

    fun setBackgroundAutomation(enabled: Boolean) {
        viewModelScope.launch { settingsRepository.setBackgroundAutomationEnabled(enabled) }
    }
}
```

```kotlin
    val ttsVoice by viewModel.ttsVoiceName.collectAsStateWithLifecycle()

        modifier = Modifier
            .fillMaxSize()
            .verticalScroll(rememberScrollState())
            .background(MaterialTheme.colorScheme.background)

        // Screen title
```
