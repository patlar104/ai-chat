```yaml
      - name: Check ktlint formatting

      - name: Run Detekt static analysis
        run: ./gradlew detekt
        continue-on-error: true

      - name: Verify no feature-to-feature dependencies

```

**File 2: `.github/scripts/check-feature-deps.sh`**

```bash
#!/usr/bin/env bash
# Belt-and-suspenders check: fails CI if any feature module build.gradle.kts
# references another :feature: module as a project dependency.
# The Gradle withDependencies hook in root build.gradle.kts catches this at sync time.
# This script catches it at the file diff level during CI for belt-and-suspenders protection.

set -e

VIOLATIONS=$(grep -rn 'project(":feature:' feature/*/build.gradle.kts 2>/dev/null || true)

if [ -n "$VIOLATIONS" ]; then
  echo "ERROR: Feature-to-feature dependency detected. Feature modules must only depend on :core:* modules."
  echo ""
  echo "Violations found:"
  echo "$VIOLATIONS"
  echo ""
  echo "Move shared types to :core:domain instead."
  exit 1
fi

echo "OK: No feature-to-feature dependencies detected."
```

```yaml
# Detekt 1.23.8, warning-only in CI (continue-on-error: true in ci.yml)
# Rules are tightened in Phase 6 hardening pass.

build:
  maxIssues: 0
  excludeCorrectable: false
  weights:
    # All severity weights set to 0 in Phase 1 — reports issues but does not fail build
    complexity: 0
    coroutines: 0
    empty-blocks: 0
    exceptions: 0
    formatting: 0
    naming: 0
    performance: 0
    potential-bugs: 0
    style: 0

config:
  validation: true
  warningsAsErrors: false

processors:
  active: true
  exclude:
    - 'DetektProgressListener'

console-reports:
  active: true
  exclude:
    - 'ProjectStatisticsReport'
    - 'ComplexityReport'
    - 'NotificationReport'
    - 'FindingsReport'
    - 'FileBasedFindingsReport'

output-reports:
  active: true

# Naming rules — permissive for Phase 1
naming:
  active: true
  ClassNaming:
    active: true
    classPattern: '[A-Z][a-zA-Z0-9]*'
  FunctionNaming:
    active: true
    functionPattern: '([a-z][a-zA-Z0-9]*)|([A-Z][a-zA-Z0-9]*)'
    # Allow capitalized names for Composable functions (e.g., VoiceScreen, AppNavHost)

complexity:
  active: true
  LongMethod:
    active: true
    threshold: 60
  LongParameterList:

    functionThreshold: 6
    constructorThreshold: 7

  active: true
  MagicNumber:
    active: true
    ignoreNumbers:
      - '-1'
      - '0'
      - '1'
      - '2'
      - '17'
      - '26'
      - '35'
```
